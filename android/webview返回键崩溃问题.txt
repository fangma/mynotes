1 解决webview返回崩溃问题
    /**
     * 解决webview返回崩溃问题(会引起finish之后的闪烁问题已在manifest处理)
     */
    @Override
    public void finish() {
        if (webView != null) {
            webView.getSettings().setBuiltInZoomControls(false);
            ViewGroup view = (ViewGroup) getWindow().getDecorView();
            view.removeAllViews();
        }
        doDestory();

        super.finish();
    }
